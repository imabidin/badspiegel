Get-ChildItem -Path . -Filter *.html | Where-Object { $_.Name -notmatch '(_de|_en)\.html$' } | ForEach-Object {
    # Basisname ohne Erweiterung
    $basename = [System.IO.Path]::GetFileNameWithoutExtension($_.Name)
    
    # Neue Dateinamen mit Suffixen
    $deFile = "${basename}_de.html"
    $enFile = "${basename}_en.html"
    
    # Originaldatei zu _de.html umbenennen
    Rename-Item $_.FullName $deFile
    
    # _de.html als _en.html kopieren
    Copy-Item $deFile $enFile
}
